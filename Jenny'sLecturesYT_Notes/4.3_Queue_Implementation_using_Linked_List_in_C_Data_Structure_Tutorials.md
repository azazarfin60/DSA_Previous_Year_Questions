## 💻 C Implementation
```c
struct node {
    int data;
    struct node *next;
};
struct node *front = NULL, *rear = NULL;

void enqueue(int data) {
    struct node *new = (struct node*)malloc(sizeof(struct node));
    new->data = data;
    new->next = NULL;
    if (rear == NULL) { front = rear = new; return; }
    rear->next = new;
    rear = new;
}

int dequeue() {
    if (front == NULL) { printf("Underflow!\n"); return -1; }
    struct node *temp = front;
    int data = temp->data;
    front = front->next;
    if (front == NULL) rear = NULL;
    free(temp);
    return data;
}
```

## 💡 Advantage over array: No overflow (dynamic), no wasted space.
