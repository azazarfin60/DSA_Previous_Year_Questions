## 💻 C Implementation
```c
struct node {
    int data;
    struct node *next;
};
struct node *front = NULL, *rear = NULL;

void enqueue(int data) {
    struct node *new = (struct node*)malloc(sizeof(struct node));
    new->data = data;
    if (front == NULL) {
        front = rear = new;
        rear->next = front;  // Circular link
    } else {
        rear->next = new;
        rear = new;
        rear->next = front;  // Maintain circular
    }
}

int dequeue() {
    if (front == NULL) return -1;
    struct node *temp = front;
    int data = temp->data;
    if (front == rear) { front = rear = NULL; }
    else { front = front->next; rear->next = front; }
    free(temp);
    return data;
}
```

## 💡 Key: `rear->next = front` maintains circularity after every operation.
