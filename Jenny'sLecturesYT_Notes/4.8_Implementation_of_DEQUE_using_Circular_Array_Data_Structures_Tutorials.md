## 💻 C Implementation
```c
#define MAX 5
int deque[MAX], front = -1, rear = -1;

void insertFront(int data) {
    if ((front == 0 && rear == MAX-1) || front == rear+1) return; // Full
    if (front == -1) front = rear = 0;
    else if (front == 0) front = MAX - 1;
    else front--;
    deque[front] = data;
}

void insertRear(int data) {
    if ((front == 0 && rear == MAX-1) || front == rear+1) return;
    if (front == -1) front = rear = 0;
    else rear = (rear + 1) % MAX;
    deque[rear] = data;
}

int deleteFront() {
    if (front == -1) return -1;
    int data = deque[front];
    if (front == rear) front = rear = -1;
    else front = (front + 1) % MAX;
    return data;
}

int deleteRear() {
    if (front == -1) return -1;
    int data = deque[rear];
    if (front == rear) front = rear = -1;
    else if (rear == 0) rear = MAX - 1;
    else rear--;
    return data;
}
```

## 💡 DEQUE supports insert/delete at **both ends** — implements both Stack and Queue.
